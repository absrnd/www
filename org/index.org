(defun create-links ()
  (let ((post-list (directory-files "~/www/absrnd/org/_posts" nil "org$"))
	(result))
    (dolist (post post-list result)
      (setq result (concat result
			   (concat "[["
				   "./_posts/" post ".html]"
				   "[" post "]]"))))))
(create-links)
#+end_src
